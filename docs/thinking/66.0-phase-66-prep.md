/**
 * Validate a captured (entityType, slug) tuple against the
 * appropriate slug-set per the Phase-63 CROSS_ENTITY_BUILD_HREF
 * routing convention. Returns true if the target resolves.
 */
export function isValidWikilinkTarget(
  ref: WikilinkReference,
  validTargets: ValidWikilinkTargets,
): boolean {
  if (ref.entityType === "paper") return validTargets.paperIds.has(ref.slug);
  if (ref.entityType === "author") return validTargets.authorSlugs.has(ref.slug);
  if (ref.entityType === "institution") return validTargets.institutionSlugs.has(ref.slug);
  // Bare slug OR unknown entity-type → problem-slug fallback
  // (mirrors CROSS_ENTITY_BUILD_HREF graceful-degradation
  // fallback to /problems/{slug} for undefined OR unknown
  // entityType).
  return validTargets.problemSlugs.has(ref.slug);
}

export interface ValidWikilinkTargets {
  problemSlugs: ReadonlySet<string>;
  paperIds: ReadonlySet<string>;
  authorSlugs: ReadonlySet<string>;
  institutionSlugs: ReadonlySet<string>;
}

### Phase-66-blocking decisions deferred to per-unit implementation

- **D-8** (Unit 66.1): Validator file placement —
  `lib/markdown/extensions/wikilinks-validator.ts` (sibling
  to `wikilinks.ts`) vs `lib/content/wikilinks-validator.ts`
  (sibling to the audit script). Lean = **`lib/markdown/
  extensions/wikilinks-validator.ts`** because the validator
  must mirror the plugin's WIKILINK_PATTERN regex + cross-
  entity routing exactly; co-locating the validator with the
  plugin keeps that consistency review-able. The audit
  consumer in `lib/content/cross-link-audit.ts` imports the
  validator across the directory boundary; no circular
  dependency because the validator does not import from
  `lib/content/`.
- **D-9** (Unit 66.1): `WIKILINK_PATTERN` re-use approach —
  re-export the constant from `wikilinks.ts` vs duplicate
  it in `wikilinks-validator.ts`. Lean = **re-export** (avoid
  drift between regex used by plugin vs validator).
- **D-10** (Unit 66.1): Validator return shape —
  `WikilinkReference[]` with `surface` field embedded vs
  caller-supplies-surface. Lean = **embed surface in
  WikilinkReference** (lets the audit consumer emit findings
  with full context without juggling separate arrays).
- **D-11** (Unit 66.1): Validator handles malformed YAML
  loads — out of scope (validator receives `markdown: string`
  + `surface: MarkdownSurface`; YAML parsing is the caller's
  responsibility). Lean = **caller-owned YAML parsing**.
- **D-12** (Unit 66.2): `wikilink-target-fk` severity —
  ERROR vs WARNING. Lean = **ERROR** (mirrors `paper-problem-
  fk` precedent for dangling FK references; consistent with
  the 404-handling intent — unresolved wikilinks should fail
  CI).
- **D-13** (Unit 66.2): Surface enumeration approach —
  hardcoded YAML field-paths per surface mapping vs
  generic schema-driven traversal. Lean = **hardcoded
  field-paths** (the 4 surfaces are stable per Phase 42
  cross-surface mapping; Phase-37 framework reserved
  `MarkdownSurface` type-union evolution as a future
  affordance; a schema-driven traversal would over-engineer
  the current scope). Validator design supports future
  expansion via the surface parameter.
- **D-14** (Unit 66.2): Test fixture strategy — add a
  `__fixtures__/` directory under `lib/content/` with
  intentional-violation YAMLs vs synthesize fixtures
  in-memory inside tests. Lean = **in-memory fixtures via
  `vi.mock` of `node:fs`** (no new filesystem state; avoids
  Velite indexing fixture-content as real content).
- **D-15** (Unit 66.1): APPEND-D-AX naming — Excel
  convention continues (D-AW → D-AX). Lean = **continue
  Excel**.
- **D-16** (Unit 66.1): Build-time-validation framing —

  axis. Lean = **audit-layer extension** (no new principal
  axis declared at Phase 66; Phase 67+ could elevate if a
  second realization ships per D-F deferrals list).
