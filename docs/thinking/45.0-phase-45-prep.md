```ts
import { visit } from "unist-util-visit";
import type { Link, PhrasingContent, Root, Text } from "mdast";

import type { MarkdownExtensionRegistry, MarkdownExtensionSet, MarkdownSurface } from "./types";

// Crossref DOI format: 10.<registrant 4-9 digits>/<suffix>. Suffix
// per https://www.crossref.org/blog/dois-and-matching-regular-expressions/.
// Phase 45 prefix-required (`doi:`-prefix) per arxiv precedent;
// bare-DOI matching deferred to Phase 46+ per ambiguity with
// numeric prose.
const DOI_PATTERN = /\bdoi:(10\.\d{4,9}\/[-._;()/:A-Za-z0-9]+?)(?=[\s.,;)]|$)/gi;

export function remarkLinkDoiIds() {
  return function transformer(tree: Root): undefined {
    visit(tree, "text", (node, index, parent) => {
      // …mirrors `remarkLinkArxivIds` exactly: scan node.value,
      // splice-replace each match with an mdast link node.
    });
    return undefined;
  };
}

export class DoiExtensionRegistry implements MarkdownExtensionRegistry {
  constructor(private readonly enabledSurfaces: ReadonlySet<MarkdownSurface>) {}

  getExtensions(surface: MarkdownSurface): MarkdownExtensionSet {
    if (this.enabledSurfaces.has(surface)) {
      return { remarkPlugins: [remarkLinkDoiIds] };
    }
    return {};
  }
}

export const PHASE_45_DEFAULT_ENABLED_SURFACES: ReadonlySet<MarkdownSurface> = new Set([
  "rationale",
]);
```

**Plugin shape**: function-declaration (factory returning a
transformer) NOT `Plugin<[], Root>` type alias — Phase 41
discovery (carried in handoff): the `Plugin<>` type alias
triggers TypeScript's overload-resolution union-narrowing
issue when chained under `unified().use(remarkParse).use(remarkLinkDoiIds)`.
**Mirrors Phase-41 arxiv shape** + `remark-gfm` / `remark-
parse` / `remark-rehype` idiom.

**Trailing-punctuation discipline**: regex uses lookahead
`(?=[\s.,;)]|$)` to bound DOI suffix matching against
sentence-terminator punctuation. This handles "see doi:10.1234/abc.def, also see…"
correctly (matches `doi:10.1234/abc.def`, NOT `doi:10.1234/abc.def,`).
The DOI suffix itself can contain periods + slashes per
Crossref — the lookahead's trailing-`.` exclusion will
correctly bound `doi:10.1234/abc.def.` (sentence-terminator
period) BUT will fail on DOIs with legitimate trailing-period
suffixes (e.g., hypothetical `doi:10.1234/abc.`). This is the
**explicit Phase-45 ship tradeoff**: most curator-prose DOI
citations end on sentence-terminator punctuation; the lookahead
favors typical-prose-correctness over edge-case-DOI-correctness.
Phase 46+ may add a stricter lookahead via curator-content
demand signal. D-3 sub-decision **D-3a**: Phase-45 ship uses
the lookahead approach. Lean = **lookahead**.
